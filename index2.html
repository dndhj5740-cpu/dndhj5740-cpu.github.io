// ==UserScript== // @name 百度自动填入脚本 // @match https://wappass.baidu.com/* // ==/UserScript== window.onload = function() { const phone = localStorage.getItem('auto_phone_val'); if (phone) { // 自动填入手机号 const input = document.querySelector('input[type="text"]'); if (input) input.value = phone; // 自动勾选协议（注意：百度协议勾选框的定位符可能因版本而变） const checkbox = document.querySelector('input[type="checkbox"]'); if (checkbox) checkbox.checked = true; // 自动点击“下一步” const nextBtn = document.querySelector('.pass-button-submit'); if (nextBtn) nextBtn.click(); // 清除缓存 localStorage.removeItem('auto_phone_val'); } };
